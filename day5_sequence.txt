4)Write a program that reads 5 random 2 digit value find the sum and avg

$ cat 5random.sh
#!/bin/bash
lower=10
sum=0
upper=99
mod=1+upper-lower
for((i=1;i<=5;i++))
        do
                random=$((lower+RANDOM%mod))
                echo $random
                sum=$((sum+random))
        done
echo "sum is :"$sum
avg=$(($sum/5))
echo "avg is :"$avg

#!/bin/bash
read -p "enter the year :" year
max=9999
min=1000
if [[ $year -ge $min && $year -le $max ]]
then

        if [[ $year%4 -eq 0 && $year%100 -ne 0 ]] || [[ $year%400 -eq 0 ]]
        then
                echo "leap year"
        else
                echo "not leap year"

else
        echo "Enter 4 digit year"
